Extrator Google Docs sem OCR/PDF. Usa `docs.documents.get`, percorre `body.content[]` concatenando texto preservando ordem (paragraphs, headings, textRuns, tables).

```js
// server/services/extractors/google-docs.js
import { getAccessToken, GoogleReauthRequiredError } from '../google-oauth.js'

function extractDocumentId(url) {
  const m = String(url).match(/\/document\/d\/([a-zA-Z0-9_-]+)/)
  if (!m) throw new Error(`Google Doc URL invalida: ${url}`)
  return m[1]
}

function walkContent(content) {
  const parts = []
  for (const el of content || []) {
    if (el.paragraph) {
      const line = (el.paragraph.elements || [])
        .map(e => e.textRun?.content || '')
        .join('')
      if (line.trim()) parts.push(line)
    } else if (el.table) {
      for (const row of el.table.tableRows || []) {
        const cells = (row.tableCells || []).map(cell => walkContent(cell.content).trim()).filter(Boolean)
        if (cells.length) parts.push(cells.join(' | '))
      }
    } else if (el.sectionBreak) {
      parts.push('')
    }
  }
  return parts.join('\n')
}

export async function extractGoogleDoc(url, { userId } = {}) {
  if (!userId) throw new Error('extractGoogleDoc requer userId')
  const documentId = extractDocumentId(url)
  const token = await getAccessToken(userId) // pode lancar GoogleReauthRequiredError
  const res = await globalThis.fetch(`https://docs.googleapis.com/v1/documents/${documentId}`, {
    headers: { Authorization: `Bearer ${token}` },
    signal: AbortSignal.timeout(60_000)
  })
  if (res.status === 401 || res.status === 403) {
    throw new GoogleReauthRequiredError(`Docs ${res.status}`, { userId, reason: 'forbidden' })
  }
  if (!res.ok) throw new Error(`Docs ${res.status}: ${await res.text().catch(() => '')}`)
  const doc = await res.json()
  const texto = walkContent(doc.body?.content)
  return { texto, imagens: [], erros: [] }
}

export default extractGoogleDoc
```

Extrator Google Slides. Passos:

1. Parse presentation ID do URL (`/presentation/d/{ID}/`).
2. Obter access_token via `getAccessToken(userId)`.
3. `GET slides.googleapis.com/v1/presentations/{id}` → estrutura.
4. Para cada slide: coletar texto (textRuns), tabelas, speakerNotes.
5. Coletar IDs de imagem por slide. Para cada imagem, baixar via `contentUrl` do Slides (campo `pageElement.image.contentUrl`) OU `drive.files.get?alt=media&mimeType=image/png` — **o research doc dita qual usar; respeitar**.
6. Para CADA imagem, enfileirar chamada GPT-4o vision com `detail=high` usando `VISION_IMAGE_PROMPT` (rate-limit via `createRateLimiter`).
7. Opcional: se o research doc mantiver `auditoria_narrativa_integral` como step final, emitir chamada GPT-4.1 JSON schema; senao retornar texto + descricoes de imagem concatenados.
8. Retorno: `{ texto, imagens: [{slideIndex, contentUrl, descricao}], erros: [] }`.

Codigo-esqueleto:

```js
// server/services/extractors/google-slides.js
import { getAccessToken, GoogleReauthRequiredError } from '../google-oauth.js'
import { createRateLimiter, withRetry } from '../../lib/rate-limiter.js'
import { VISION_IMAGE_PROMPT, AUDITORIA_NARRATIVA_PROMPT } from './openai-prompts.js'

const visionLimiter = createRateLimiter({
  type: 'concurrent-rpm',
  maxConcurrent: parseInt(process.env.VISION_MAX_CONCURRENT || '3', 10),
  maxRpm: parseInt(process.env.VISION_MAX_RPM || '60', 10)
})

function extractPresentationId(url) {
  const m = String(url).match(/\/presentation\/d\/([a-zA-Z0-9_-]+)/)
  if (!m) throw new Error(`Google Slides URL invalida: ${url}`)
  return m[1]
}

function collectTextFromPageElement(pe) {
  // textBox / shape com textRuns
  const parts = []
  const elements = pe.shape?.text?.textElements || pe.elementGroup?.children?.flatMap(c => c.shape?.text?.textElements || []) || []
  for (const te of elements) {
    if (te.textRun?.content) parts.push(te.textRun.content)
  }
  if (pe.table) {
    for (const row of pe.table.tableRows || []) {
      for (const cell of row.tableCells || []) {
        for (const te of cell.text?.textElements || []) {
          if (te.textRun?.content) parts.push(te.textRun.content)
        }
      }
    }
  }
  return parts.join('')
}

async function describeImage(contentUrl, token) {
  await visionLimiter.acquire()
  try {
    return await withRetry(async () => {
      const res = await globalThis.fetch('https://api.openai.com/v1/chat/completions', {
        method: 'POST',
        headers: {
          Authorization: `Bearer ${process.env.OPENAI_API_KEY}`,
          'Content-Type': 'application/json'
        },
        body: JSON.stringify({
          model: 'gpt-4o',
          messages: [{
            role: 'user',
            content: [
              { type: 'text', text: VISION_IMAGE_PROMPT },
              { type: 'image_url', image_url: { url: contentUrl, detail: 'high' } }
            ]
          }]
        }),
        signal: AbortSignal.timeout(120_000)
      })
      if (!res.ok) throw new Error(`vision ${res.status}: ${await res.text().catch(() => '')}`)
      const json = await res.json()
      return json.choices?.[0]?.message?.content || ''
    })
  } finally {
    visionLimiter.release()
  }
}

export async function extractGoogleSlides(url, { userId } = {}) {
  if (!userId) throw new Error('extractGoogleSlides requer userId')
  const presentationId = extractPresentationId(url)
  const token = await getAccessToken(userId)
  const res = await globalThis.fetch(`https://slides.googleapis.com/v1/presentations/${presentationId}`, {
    headers: { Authorization: `Bearer ${token}` },
    signal: AbortSignal.timeout(60_000)
  })
  if (res.status === 401 || res.status === 403) {
    throw new GoogleReauthRequiredError(`Slides ${res.status}`, { userId, reason: 'forbidden' })
  }
  if (!res.ok) throw new Error(`Slides ${res.status}: ${await res.text().catch(() => '')}`)
  const pres = await res.json()

  const textoParts = []
  const imagens = []
  const erros = []

  for (const [slideIdx, slide] of (pres.slides || []).entries()) {
    const slideTexts = []
    for (const pe of slide.pageElements || []) {
      if (pe.shape?.text || pe.table) slideTexts.push(collectTextFromPageElement(pe))
      if (pe.image?.contentUrl) {
        try {
          const descricao = await describeImage(pe.image.contentUrl, token)
          imagens.push({ slideIndex: slideIdx, contentUrl: pe.image.contentUrl, descricao })
          slideTexts.push(`[Imagem slide ${slideIdx + 1}] ${descricao}`)
        } catch (err) {
          erros.push({ slideIndex: slideIdx, mensagem: err.message })
        }
      }
    }
    // Speaker notes
    const notes = slide.slideProperties?.notesPage?.pageElements
      ?.flatMap(pe => pe.shape?.text?.textElements || [])
      ?.map(te => te.textRun?.content || '').join('') || ''
    if (notes.trim()) slideTexts.push(`[Notas slide ${slideIdx + 1}] ${notes}`)

    textoParts.push(`## Slide ${slideIdx + 1}\n${slideTexts.join('\n')}`)
  }

  return { texto: textoParts.join('\n\n'), imagens, erros }
}

export default extractGoogleSlides
```
