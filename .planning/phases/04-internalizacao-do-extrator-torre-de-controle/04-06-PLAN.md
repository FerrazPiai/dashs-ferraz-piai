```js
// client/stores/googleConnection.js
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

export const useGoogleConnectionStore = defineStore('googleConnection', () => {
  const loading = ref(false)
  const connected = ref(false)
  const email = ref(null)
  const connectedAt = ref(null)
  const scopesOk = ref(false)
  const lastCheckedAt = ref(null)
  const error = ref(null)

  const statusLabel = computed(() => {
    if (!connected.value) return 'Nao conectado'
    if (!scopesOk.value) return 'Escopos desatualizados'
    return 'Ativo'
  })

  async function fetchStatus({ force = false } = {}) {
    if (!force && lastCheckedAt.value && Date.now() - lastCheckedAt.value < 30_000) return
    loading.value = true
    error.value = null
    try {
      const res = await fetch('/api/google/status', { credentials: 'include' })
      if (!res.ok) throw new Error(`status ${res.status}`)
      const data = await res.json()
      connected.value = !!data.connected
      email.value = data.email || null
      connectedAt.value = data.connected_at || null
      scopesOk.value = !!data.scopes_ok
      lastCheckedAt.value = Date.now()
    } catch (err) {
      error.value = err.message
    } finally {
      loading.value = false
    }
  }

  async function startConnect({ resumeUrl } = {}) {
    const res = await fetch('/api/google/connect-drive/start', {
      method: 'POST',
      credentials: 'include',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ resume_url: resumeUrl || window.location.pathname + window.location.search })
    })
    if (!res.ok) throw new Error(`start ${res.status}`)
    const { url } = await res.json()
    window.location.href = url
  }

  async function disconnect() {
    const res = await fetch('/api/google/disconnect', { method: 'POST', credentials: 'include' })
    if (!res.ok) throw new Error(`disconnect ${res.status}`)
    connected.value = false
    email.value = null
    connectedAt.value = null
    scopesOk.value = false
  }

  return { loading, connected, email, connectedAt, scopesOk, statusLabel, error, fetchStatus, startConnect, disconnect }
})
```
