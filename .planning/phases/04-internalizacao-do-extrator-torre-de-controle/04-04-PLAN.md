```js
// server/services/extractors/miro.js
const MIRO_BASE = 'https://api.miro.com/v2'

function extractBoardId(url) {
  const m = String(url).match(/\/app\/board\/([A-Za-z0-9_=-]+)/)
  if (!m) throw new Error(`Miro board URL invalida: ${url}`)
  // Miro boards as vezes tem `=` no final; manter como esta
  return m[1]
}

async function fetchAllItems(boardId, token) {
  const items = []
  let cursor = null
  do {
    const url = new URL(`${MIRO_BASE}/boards/${boardId}/items`)
    url.searchParams.set('limit', '50')
    if (cursor) url.searchParams.set('cursor', cursor)
    const res = await globalThis.fetch(url.toString(), {
      headers: { Authorization: `Bearer ${token}`, Accept: 'application/json' },
      signal: AbortSignal.timeout(60_000)
    })
    if (!res.ok) throw new Error(`Miro items ${res.status}: ${await res.text().catch(() => '')}`)
    const page = await res.json()
    items.push(...(page.data || []))
    cursor = page.cursor || null
  } while (cursor)
  return items
}

export async function extractMiro(url, _opts = {}) {
  const token = process.env.MIRO_TOKEN
  if (!token) throw new Error('MIRO_TOKEN nao configurado')
  const boardId = extractBoardId(url)
  const items = await fetchAllItems(boardId, token)

  const textoParts = []
  const imagens = []
  const erros = []

  for (const it of items) {
    switch (it.type) {
      case 'sticky_note':
      case 'text':
      case 'shape':
        textoParts.push((it.data?.content || '').replace(/<[^>]+>/g, '').trim())
        break
      case 'image':
        if (it.data?.imageUrl || it.data?.title) {
          imagens.push({ id: it.id, url: it.data?.imageUrl, title: it.data?.title })
          textoParts.push(`[Imagem ${it.data?.title || it.id}]`)
        }
        break
      default:
        // Preserve o tipo em texto para nao perder sinal
        if (it.data?.title) textoParts.push(`[${it.type}] ${it.data.title}`)
    }
  }

  return {
    texto: textoParts.filter(Boolean).join('\n'),
    imagens,
    erros,
    _meta: { totalItens: items.length }
  }
}

export default extractMiro
```
